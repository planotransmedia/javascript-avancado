    function JSClock() {
        var time = new Date();
        var hora = time.getHours();
        var minuto = time.getMinutes();
        var segundo = time.getSeconds();
        var temp = "" + ((hora > 12) ? hora - 12 : hora);
        if (hora == 0)
            temp = "12";
        temp += ((minuto < 10) ? ":0" : ":") + minuto;
        temp += ((segundo < 10) ? ":0" : ":") + segundo;
        temp += (hora >= 12) ? " P.M." : " A.M.";
        return temp;
    }

    function Eletrodomestico() {
        this.ligado = false;
        this.ligar = function() {
            this.ligado = true;

            this.ligado = false;
